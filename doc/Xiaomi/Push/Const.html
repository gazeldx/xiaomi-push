def build_uri(uri)
  version =
    if uri.start_with?('stats', 'trace', 'alias', 'reg_id') || uri == 'topic/all'
      # 获取消息的统计数据/追踪消息状态/某个用户目前设置的所有 Alias 和订阅的所有 Topic
      'v1'
    elsif uri.start_with?('message')
      # 发送消息支持多包使用 v3 版本
      'v3'
    else
      'v2'
    end

  File.join(@base_url, version, uri)
end
